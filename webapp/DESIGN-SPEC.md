## Technical Notes

### State Management

The AI GM needs to track:

```typescript
interface GameState {
  // Character
  character: {
    name: string;
    background: Background;
    attributes: { grit: number; reflex: number; wits: number; nerve: number };
    skills: Record<SkillName, number>;
    wounds: { bruised: number; bleeding: number; broken: number; critical: boolean };
    stress: number;
    guts: number;
    gutsEarnedThisSession: number;
    inventory: Item[];
    weapons: Weapon[];
    armor: Armor | null;
    carryingCapacity: number;
  };
  
  // World
  world: {
    day: number;
    time: string; // "night", "dawn", "day", "dusk"
    location: Location;
    threat: number;
    threatState: "safe" | "noticed" | "investigating" | "encounter" | "swarm";
    ambientThreat: number;
  };
  
  // Party
  party: NPC[];
  
  // Story
  story: {
    objectives: Objective[];
    events: StoryEvent[];
    relationships: Record<string, Relationship>;
  };
  
  // Session
  session: {
    messageHistory: Message[];
    rollHistory: Roll[];
    startTime: Date;
    combatState: CombatState | null;
  };
}
```
